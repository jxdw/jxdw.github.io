import (

"os"
"log"
"io/ioutil"
"encoding/json"
//"encoding/csv"
"fmt"
"encoding/csv"
//"strconv"
"strconv"

)

type Mydata struct {
Data []FundData json:"data"
}
type FundData struct {
FeeType string json:"feeType"
TotalFee float64 json:"totalFee"
Tano string json:"tano"
ManagerCode string json:"managerCode"
FundCode string json:"fundCode"
AccuVol float64 json:"accuVol"
AccuAmount float64 json:"accuAmount"
}
func main(){
f, err := os.Open(“data.json”)
if err != nil {
log.Fatal(err)
}
ret, err := ioutil.ReadAll(f)
if err != nil {
log.Fatal(err)
}
var mydata=Mydata{}
err=json.Unmarshal(ret,&mydata)
if err!=nil {
fmt.Println(err)
}
var datavalue []FundData =mydata.Data
fmt.Println(len(datavalue))
file, err := os.Create(“test.xls”)
if err != nil {
panic(err)
}
defer func(file *os.File) { file.Close() }(file)
file.WriteString(“\xEF\xBB\xBF”) // 写入UTF-8 BOM
w := csv.NewWriter(file)
w.Write([]string{“栏目1”, “栏目2”, “栏目3”,”栏目4”,”栏目5”,”栏目6”,”栏目7”})
for _,value:=range datavalue{
fee:=strconv.FormatFloat(value.TotalFee, ‘f’, -1, 64)
vol:=strconv.FormatFloat(value.AccuVol, ‘f’, -1, 64)
amount:=strconv.FormatFloat(value.AccuAmount, ‘f’, -1, 64)
w.Write([]string{value.FeeType,fee,value.ManagerCode,value.Tano,value.FundCode,vol,amount})
}
w.Flush()
}
